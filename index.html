let currentRotation = -35;

let currentZoom = 1;

// =============== CAMERA CONTROLS ===============

function rotateScene() {

    currentRotation -= 15;

    if(currentRotation < -60) currentRotation = -10;

    updateTransform();

}


function zoomIn() {

    if(currentZoom < 1.3){ currentZoom += 0.15; updateTransform(); }

}


function zoomOut() {

    if(currentZoom > 0.7){ currentZoom -= 0.15; updateTransform(); }

}


function updateTransform() {

    scene.style.transform = `rotateX(55deg) rotateZ(${currentRotation}deg) scale(${currentZoom})`;

}


// =============== MODAL LOGIC ===============

function openModal(buildingEl) {

    // Add selection animation to building

    buildingEl.classList.add('selected');

    

    // Parse task data from dataset

    const taskData = JSON.parse(buildingEl.dataset.task);

    

    // Populate modal

    const typeLabels = {

        'home': 'ЖИЛОЙ КОМПЛЕКС',

        'partner': 'ПАРТЁР',

        'task': 'ЗАДАНИЕ НЕДЕЛИ',

        'event': 'СОБЫТИЕ'

    };

    

    document.getElementById('modalTypeBadge').textContent = typeLabels[taskData.type];

    document.getElementById('modalTypeBadge').style.background = taskData.type === 'home' ? '#FF6B35' : 

                                                              taskData.type === 'partner' ? '#3498DB' : 

                                                              '#4CAF50';

    

    document.getElementById('modalBuildingName').textContent = taskData.name;

    document.getElementById('modalLocation').innerHTML = `${taskData.location}`;

    // Show modal

    setTimeout(() => {

        document.getElementById('modalOverlay').classList.add('active');

    }, 300);

}


function closeModal(event) {

    document.getElementById('modalOverlay').classList.remove('active');

}

function launchConfetti() {

    const container = document.getElementById('confettiContainer');

    container.innerHTML = '';

    

    const colors = ['#FF6B35', '#3498DB', '#4CAF50', '#F39C12', '#9B59B6', '#E74C3C'];

    

    for(let i = 0; i < 60; i++) {

        const piece = document.createElement('div');

        piece.className = 'confetti-piece';

        piece.style.left = Math.random() * 100 + '%';

        piece.style.background = colors[Math.floor(Math.random() * colors.length)];

        piece.style.width = Math.random() * 8 + 6 + 'px';

        piece.style.height = Math.random() * 8 + 6 + 'px';

        piece.style.borderRadius = Math.random() > 0.5 ? '50%' : '2px';

        piece.style.animationDuration = Math.random() * 2 + 1.5 + 's';

        piece.style.animationDelay = Math.random() * 0.5 + 's';

        container.appendChild(piece);

    }

    

    setTimeout(() => { container.innerHTML = ''; }, 3500);

}
